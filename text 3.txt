self.addEventListener("install", e => {
  e.waitUntil(
    caches.open("water-calc").then(cache => {
      return cache.addAll(["./"]);
    })
  );
});

self.addEventListener("fetch", event => {
  event.respondWith(
    caches.match(event.request).then(response => {
      return response || fetch(event.request);
    })
  );
});
