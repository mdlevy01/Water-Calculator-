<script>
let unit="F";
const KG_TO_LB=2.20462;

function kgToLb(){
 let v=parseFloat(kg.value);
 lb.value=(v*KG_TO_LB).toFixed(2);
}
function lbToKg(){
 let v=parseFloat(lb.value);
 kg.value=(v/KG_TO_LB).toFixed(2);
}

function cToF(c){return (c*9/5)+32;}
function fToC(f){return (f-32)*5/9;}

function calcTemp(){
 let f=parseFloat(flour.value);
 let r=parseFloat(room.value);
 let fr=parseFloat(friction.value);
 let d=parseFloat(ddt.value);
 let pVal=pref.value;
 let p=pVal?parseFloat(pVal):0;

 if(unit==="C"){
  f=cToF(f); r=cToF(r); fr=cToF(fr); d=cToF(d);
  if(pVal) p=cToF(p);
 }

 let factors=pVal?4:3;
 let result=(d*factors)-(f+r+fr+p);
 let final=unit==="C"?fToC(result):result;

 resultEl.innerText="Water Temp: "+final.toFixed(1)+"°"+unit;
}

function toggleUnit(){
 unit=unit==="F"?"C":"F";
 resultEl.innerText="";
}
</script>
